CREATE TABLE ITEM (
ITEM_ID NUMBER PRIMARY KEY,
BANK_ACC_NO VARCHAR(30),
TXN_REFERENCE VARCHAR(50),
TXN_AMOUNT NUMBER(12,4),
DRORCR VARCHAR(1),
TXN_CCY VARCHAR(3),
STMT_DATE DATE,
SYSTEM_ENTRY_DATE TIMESTAMP,
VALUE_DATE DATE,
ITEM_TYPE VARCHAR(20),
TXN_STOCK_CODE VARCHAR(12),
STOCK_ID NUMBER,
QUANTITY NUMBER(12,4),
PRICE NUMBER(12,4),
CONSTRAINT FK_BANK_ACC_NO FOREIGN KEY (BANK_ACC_NO) REFERENCES BANK_ACCOUNTS(BANK_ACC_NO)
);
